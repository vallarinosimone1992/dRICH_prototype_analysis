cmake_minimum_required(VERSION 3.5)

project(dRICH_reco)

find_package(ROOT 6.16 CONFIG REQUIRED)

include("${ROOT_USE_FILE}")

separate_arguments(ROOT_EXE_LINKER_FLAGS)


add_library(dRICH_lib
      STATIC
      lib/definition.h
      lib/utility.cxx
      lib/computing.cxx
      lib/getChannel.cxx
      lib/photoDetPosition.cxx
      lib/selection.cxx
      lib/correction.cxx
      lib/fillMAPS.cxx
      lib/integrate.cxx
      lib/readData.cxx
      lib/tracking.cxx
       lib/drawing.cxx
      lib/writeHeaderText.cxx
)

target_include_directories(dRICH_lib
      PUBLIC 
      ${PROJECT_SOURCE_DIR}/lib
)


add_executable(reco
      plugin/dRICH_reco.cxx
)

target_link_libraries(reco
      PRIVATE
      dRICH_lib
      PUBLIC
      ${ROOT_LIBRARIES}
      ${ROOT_EXE_LINKER_FLAGS}
)


add_executable(ana
      plugin/analysis.cxx
)


target_link_libraries(ana
      PRIVATE
      dRICH_lib
      PUBLIC
      ${ROOT_LIBRARIES}
      ${ROOT_EXE_LINKER_FLAGS}
)

add_executable(mon
     plugin/monitor.cxx
)

target_link_libraries(mon
      PRIVATE
      dRICH_lib
      PUBLIC
      ${ROOT_LIBRARIES}
      ${ROOT_EXE_LINKER_FLAGS}
)
